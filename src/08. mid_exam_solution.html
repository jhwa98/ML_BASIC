def loss_mse(w, x, t):
    y = w[0] + w[1]*x + w[2]*(x**2)
    val = 0.0    
    for n in range(len(x)):        
        val = val + np.square(t[n]-y[n])
    val = val / x.shape[0]
    return val

def logistic2(x0, x1, w):
    y = 1 / (1 + np.exp(-(w[0] * x0 + w[1] * x1 + w[2])))
    return y

def cost_logistic2(w, x, t):
    X_n = x.shape[0]
    y = logistic2(x[:, 0], x[:, 1], w)    
    cee = 0
    for n in range(len(y)):
        cee = cee - (t[n, 0] * np.log(y[n]) + (1 - t[n, 0]) * np.log(1 - y[n]))        
    cee = cee / X_n
    return cee

def show_line(W):
